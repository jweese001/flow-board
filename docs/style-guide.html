Tailwind v4 CSS Reference

Copy this @theme block into your index.css file.

/* Tailwind v4 @theme block */
@import "tailwindcss";

@theme {
  /* Backgrounds */
  --color-bg-darkest: #0a0b0f;
  --color-bg-dark: #12141a;
  --color-bg-panel: #181a22;
  --color-bg-elevated: #1e212b;
  --color-bg-hover: #252836;

  /* Text */
  --color-text-primary: #e8eaed;
  --color-text-secondary: #9ca3af;
  --color-text-muted: #6b7280;
  --color-text-placeholder: #4b5563;

  /* Borders */
  --color-border-subtle: rgba(255, 255, 255, 0.06);
  --color-border-medium: rgba(255, 255, 255, 0.1);
  --color-border-strong: rgba(255, 255, 255, 0.15);

  /* Node Colors */
  --color-node-character: #3b82f6;
  --color-node-setting: #10b981;
  --color-node-prop: #f59e0b;
  --color-node-style: #a855f7;
  --color-node-extras: #64748b;
  --color-node-shot: #ec4899;
  --color-node-outfit: #06b6d4;
  --color-node-action: #f97316;
  --color-node-negative: #f43f5e;
  --color-node-params: #14b8a6;
  --color-node-edit: #6b7280;
  --color-node-output: #ef4444;

  /* Accents */
  --color-accent-primary: #6366f1;
  --color-success: #22c55e;
  --color-warning: #eab308;
  --color-error: #ef4444;

  /* Typography */
  --font-display: 'Outfit', sans-serif;
  --font-mono: 'JetBrains Mono', monospace;

  /* Sizing */
  --sidebar-width: 280px;
  --node-width: 220px;
  --node-width-lg: 280px;

  /* Shadows */
  --shadow-node: 0 4px 20px rgba(0, 0, 0, 0.5);
  --shadow-panel: 0 0 40px rgba(0, 0, 0, 0.5);
}
